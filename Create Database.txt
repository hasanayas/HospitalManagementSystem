CREATE TABLE Hasta(

	HastaNo NUMBER GENERATED by default on null as IDENTITY,
	TcNo CHAR(11) NOT NULL UNIQUE,
	Ad VARCHAR2(30) NOT NULL,
	Soyad VARCHAR2(30) NOT NULL,
	Cinsiyet CHAR(1) NOT NULL,
	DogumTarihi DATE NOT NULL,
	TelefonNo CHAR(11) NOT NULL UNIQUE,
	KanGrubu CHAR(4) NOT NULL,
	Boy int NOT NULL,
	Kilo int NOT NULL,
    DoktorNo NUMBER NOT NULL,
    ReceteNo NUMBER NOT NULL,
    MuayeneNo NUMBER NOT NULL,
    SekreterNo NUMBER NOT NULL,
    CONSTRAINT ck_cinsiyetHasta 
        CHECK(Cinsiyet IN('E','K')),
    CONSTRAINT pk_HastaNo PRIMARY KEY (HastaNo),
    CONSTRAINT fk_DoktorNo
        FOREIGN KEY (DoktorNo) REFERENCES Doktor (DoktorNo),
    CONSTRAINT fk_ReceteNo
        FOREIGN KEY (ReceteNo) REFERENCES Recete (ReceteNo),
    CONSTRAINT fk_MuayeneNo
        FOREIGN KEY (MuayeneNo) REFERENCES Muayene (MuayeneNo),
    CONSTRAINT fk_SekreterNo
        FOREIGN KEY (SekreterNo) REFERENCES Sekreter (SekreterNo)

);

CREATE TABLE Doktor(

	DoktorNo NUMBER GENERATED by default on null as IDENTITY,
	TcNo CHAR(11) NOT NULL UNIQUE,
	Ad VARCHAR2(30) NOT NULL,
	Soyad VARCHAR2(30) NOT NULL,
	DogumTarihi DATE NOT NULL,
	TelefonNo CHAR(11) NOT NULL UNIQUE,
	Cinsiyet CHAR(1) NOT NULL,
	Bolumu VARCHAR2(15) NOT NULL,
    CONSTRAINT ck_cinsiyetDok
		CHECK(Cinsiyet IN('E','K')),
    CONSTRAINT pk_DoktorNo PRIMARY KEY (DoktorNo)

);

CREATE TABLE Sekreter(

	SekreterNo NUMBER GENERATED by default on null as IDENTITY,
	TcNo CHAR(11) NOT NULL UNIQUE,
	Ad VARCHAR2(30) NOT NULL,
	Soyad VARCHAR2(30) NOT NULL,
	DogumTarihi DATE NOT NULL,
	TelefonNo CHAR(11) NOT NULL UNIQUE,
	Cinsiyet CHAR(1) NOT NULL,
    CONSTRAINT ck_CinsiyetSek
		CHECK(Cinsiyet IN('E','K')),
    CONSTRAINT pk_SekreterNo PRIMARY KEY (SekreterNo)

);

CREATE TABLE Muayene(

	MuayeneNo NUMBER GENERATED by default on null as IDENTITY,
	MuayeneTarihi DATE NOT NULL,
	HastaSikayeti VARCHAR2(500) NOT NULL,
    CONSTRAINT pk_MuayeneNo PRIMARY KEY (MuayeneNo)
    

);

CREATE TABLE Recete(

	ReceteNo NUMBER GENERATED by default on null as IDENTITY,
	IlacAdi VARCHAR2(50) NOT NULL,
	IlacTuru VARCHAR2(25) NOT NULL,
	PaketSayisi NUMBER NOT NULL,
	KullanimSekli VARCHAR2(200) NOT NULL,
    CONSTRAINT pk_ReceteNo PRIMARY KEY (ReceteNo)
);
